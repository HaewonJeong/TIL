### useEffect
```

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <!-- CDN -->
    <script crossorigin src="https://unpkg.com/react@18/umd/react.development.js"></script>
    <script crossorigin src="https://unpkg.com/react-dom@18/umd/react-dom.development.js"></script> 
    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

</head>
<body>
    <div id="root"></div>

    <script type="text/babel">
        
        const {useState, useEffect} = React;
        console.log(React); //hook 종류 확인 가능

        function Counter() { //함수 컴포넌트 생성
            const [count, setCount] = useState(0);
            const [otherCount, setOtherCount] = useState(0);


            /*
            -기본적으로 useEffect는 매 렌더링 이후에 실행되지만, 의존성 배열을 통해 실행 시점을 제어할 수 있다.
            -빈 배열 ([]): 컴포넌트가 처음 마운트될 때만 useEffect가 실행되고, 이후에는 재실행되지 않는다. 이를 통해 초기화 작업 등을 수행할 수 있다.
            -배열 생략 (undefined): 배열을 생략하면, 컴포넌트가 렌더링될 때마다 useEffect가 실행된다.
            -특정 값 포함 ([stateA, propB]): 배열에 포함된 값들이 변경될 때만 useEffect가 실행된다. 이로 인해 필요한 경우에만 부수 효과(Side Effect)가 발생하도록 제어할 수 있다.
            */

            //useEffect는 첫 마운트 직후에는 무조건 한번 실행 된다.

            //1. 의존성 배열이 생략된 형태 : 매 랜더링 마다 실행
            //==>의존성 배열이 없으면 매번 실행
            useEffect( ()=>{
                //렌더링 '완료된 후' 실행된다.
                console.log('렌더링 될때마다')
                document.title = `넌 ${count} 번 클릭했어`;}
            )

            //2. 빈 의존성 배열 [] : Mount(첫 렌더링) 시에만 실행
            //주로 초기 데이터 로딩(API 호출) 등에 사용
            //==>의존성 배열이 빈 배열이면 처음 1번 실행
            useEffect( () => {
                console.log('컴포넌트가 처음 화면에 나타남(Mount)');
            }, [] );
            
            //3. 의존성 배열에 특정 값 지정 [count] : 해당 값이 변경될 때만 실행
            //==>의존성 배열에 count 넣으면 (count 바뀔 때만 실행)
            useEffect( () => {
                console.log(`count 값이 ${count}로 변경되어, 이 useEffect가 실행`);
            }, [count] ); 

            console.log('Counter 컴포넌트 렌더링')

            //document.title = `넌 ${count} 번 클릭했어`;
            //잘 동작하는것 같지만, 리액트에서는 나쁜 코드. document (실제 DOM)에 직접 접근

            return(
                <div>
                    <h2>단순 카운터</h2>
                    <p>{count}번 클릭했습니다.</p>
                    <button onClick={ () => setCount(count+1)}>클릭</button>
                    <p>Other Count: {otherCount}</p>
                    <button onClick={ () => setOtherCount(otherCount + 1)}>클릭</button>
                    {/*state가 바뀔때마다 리렌더링*/}
                </div>

            )
        }

        function Timer(){

            // 1) 실행할 코드 (effect)
            useEffect( () => {
                console.log('타이머가 시작되었습니다.(Mount)');

                const timerId = setInterval( ()=> {
                    console.log('...1초 경과...');
                }, 1000);

            //Cleanup 함수: useEffect가 반환하는 함수
            //컴포넌트가 화면에서 사라지기 직전(Unmount)에 호출 된다.
            //return에서 Cleanup을 반환

            //2) 정리할 코드 (cleanup)
                //Cleanup 함수 실행 시점: 
                //1) 컴포넌트가 화면에서 사라질 때 (Unmount)
                //2) useEffect가 다시 실행되기 직전
                //count가 바뀌면 effect를 다시 실행해야 하니까
                //그 전에 기존 effect를 정리하기 위해 cleanup 실행함.
            return () => {
                    console.log('타이머를 정리합니다.(Unmount)');
                    clearInterval(timerId); //생성했던 타이머 제거
                }
            }, [] );

            return (
                <h3>타이머가 동작 중입니다. (콘솔 확인)</h3>
            );
        }

        function App(){
            const [showTimer, setShowTimer] = useState(true); //showTimer를 boolean 값으로 선언

            return(
                <>
                    <Counter />
                    <hr />
                    <button onClick={ () => setShowTimer(!showTimer)}>타이머 보이기/숨기기</button>
                    {showTimer && <Timer />}
                    {/*showtimer가 true라면 보여주고, 아니면 안보여 준다.
                     앞이 True면 뒤를 실행, 앞이 flase를 뒤를 무시하고 false 반환*/}


                </>
            )
        }
        
        ReactDOM.createRoot(document.getElementById('root')).render(<App />);

    </script>
</body>
</html>
```
