```
package com.ohgiraffers.section01.javaconfig;

import org.apache.ibatis.datasource.pooled.PooledDataSource;
import org.apache.ibatis.mapping.Environment;
import org.apache.ibatis.session.SqlSession;
import org.apache.ibatis.session.SqlSessionFactory;
import org.apache.ibatis.session.SqlSessionFactoryBuilder;
import org.apache.ibatis.transaction.jdbc.JdbcTransactionFactory;
import org.apache.ibatis.session.Configuration;

public class Application {

    private static String DRIVER = "com.mysql.cj.jdbc.Driver";
    private static String URL = "jdbc:mysql://localhost/menudb";
    private static String USER = "ohgiraffers";
    private static String PASSWORD = "ohgiraffers";

    public static void main(String[] args){

        /*1. 환경(Environment) 설정
        *
        * 1)트랜잭션 매니저 선택:
        *   JdbcTransactionFactory -> 개발자가 commit/rollback을 직접 호출
        *   ManagedTransactionFactory -> WAS나 Spring 같은 외부 환경이 트랜잭션 관리
        * 2)DataSource 선택:
        *   PooledDataSource -> 커넥션 풀(Pool) 사용 -> 실무 표준
        *   UnpooledDataSource -> 매번 새 연결 생성 -> 테스트용
        * */
        Environment environment = new Environment(
                "dev",                            //환경 ID
                new JdbcTransactionFactory(),         //트랜잭션 매니저
                new PooledDataSource(DRIVER, URL, USER, PASSWORD)  //DB 연결 정보 + 커넥션 풀
        );

        /*2. Configuration 생성 + Mapper 등록*/
        Configuration configuration = new Configuration(environment);
        configuration.addMapper(Mapper.class); // "이 인터페이스에 Mybatis가 관리하게 등록한다.

        /*3. SqlSessionFactory 생성*/
        // sqlSessionFactory : SqlSession을 찍어내는 공장. 애플리케이션 실행 중 딱 하나만 만든다.
        SqlSessionFactory sqlSessionFactory = new SqlSessionFactoryBuilder().build(configuration);

        /*4. SqlSession 열기*/
        //1) false: auto-commit 비활성화 -> 직접 commit() 또는 rollback() 호출
        //2) true: auto-commit 활성화 -> 쿼리 실행 즉시 자동 반영
        SqlSession sqlsession = sqlSessionFactory.openSession(false);

        /*5. Mapper를 가져와서 쿼리 실행*/
        // getMapper() : Configuration에 등록된 Mapper 인터페이스의 구현 객체를 반환
        // MyBatis가 내부적으로 동적 프록시로 구현체를 생성
        Mapper mapper = sqlsession.getMapper(Mapper.class);
        java.util.Date date = mapper.selectSysdate();

        System.out.println("DB 서버 현재 날짜: " + date);

        sqlsession.close();

    }


}
```

#### mybatis-config.xml
- MyBatis의 전역 설정 파일로, 데이터베이스 연결 정보와 Mapper 파일 위치를 관리
	- transactionManager (JDBC) : 개발자가 직접 commit() / rollback()을 수행한다.
	- dataSource (POOLED) : 커넥션 풀을 사용하여 DB 연결을 재사용한다.
	- mapper resource : SQL이 작성된 Mapper XML 파일을 등록한다.
```
<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE configuration
        PUBLIC "-//mybatis.org//DTD Config 3.0//EN"
        "https://mybatis.org/dtd/mybatis-3-config.dtd">
<configuration>
    <environments default="dev">
        <environment id="dev">
            <transactionManager type="JDBC"/>
            <dataSource type="POOLED">
                <property name="driver" value="com.mysql.cj.jdbc.Driver"/>
                <property name="url" value="jdbc:mysql://localhost/menudb"/>
                <property name="username" value="ohgiraffers"/>
                <property name="password" value="ohgiraffers"/>
            </dataSource>
        </environment>
    </environments>
    <mappers>
        <mapper resource="com/ohgiraffers/section01/xmlconfig/menu-mapper.xml"/>
    </mappers>
</configuration>
```

#### menu-mapper.xml
	- 실제 SQL을 작성하는 Mapper 파일 이다.
 	- ResultMap : DB 컬럼명과 DTO 필드명이 다를 경우 매핑 규칙을 정의한다. 이를 통해 조회 결과를 MenuDTO 객체로 자동 변환할 수 있다.
```
<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "https://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="MenuMapper">
    <!-- 조회한 컬럼과 DTO를 매핑시키기 위한 설정
        컬럼은 스네이크, 필드는 소문자 카멜 -> 자동매핑 불가
        <id> = PK 컬럼, <result> = 일반 컬럼 -->
    <resultMap id="menuResultMap" type="com.ohgiraffers.section01.xmlconfig.MenuDTO">
        <id property="code" column="menu_code"/>
        <result property="name" column="menu_name"/>
        <result property="price" column="menu_price"/>
        <result property="categoryCode" column="category_code"/>
        <result property="orderableStatus" column="orderable_status"/>
    </resultMap>

    <select id="selectAllMenu" resultMap="menuResultMap">
        SELECT menu_code
            ,menu_name
            ,menu_price
            ,category_code
            ,orderable_status
        FROM tbl_menu
        WHERE orderable_status = "Y"
        ORDER BY menu_code
    </select>

    <select id="selectMenuByCode" parameterType="_int" resultMap="menuResultMap">
        SELECT menu_code
            , menu_name
            , menu_price
            , category_code
            , orderable_status
        FROM tbl_menu
        WHERE orderable_status = "Y"
        AND menu_code = #{code}
    </select>

    <insert id="insertMenu" parameterType="com.ohgiraffers.section01.xmlconfig.MenuDTO">
         INSERT INTO tbl_menu ( menu_name, menu_price, category_code, orderable_status )
         VALUES ( #{name}, #{price}, #{categoryCode}, 'Y')
    </insert>

</mapper>
```

# MyBatis XML 설정 및 동작 구조

## 1. mybatis-config.xml

MyBatis의 전역 설정 파일로, 데이터베이스 연결 정보와 Mapper 파일 등록을 담당한다.

```xml
<configuration>
    <environments default="dev">
        <environment id="dev">
            <transactionManager type="JDBC"/>
            <dataSource type="POOLED">
                <property name="driver" value="com.mysql.cj.jdbc.Driver"/>
                <property name="url" value="jdbc:mysql://localhost/menudb"/>
                <property name="username" value="ohgiraffers"/>
                <property name="password" value="ohgiraffers"/>
            </dataSource>
        </environment>
    </environments>

    <mappers>
        <mapper resource="com/ohgiraffers/section01/xmlconfig/menu-mapper.xml"/>
    </mappers>
</configuration>
```

### TransactionManager

```xml
<transactionManager type="JDBC"/>
```

트랜잭션을 관리하는 설정이다.

* JDBC 방식 사용
* 개발자가 직접 `commit()` / `rollback()` 호출
* Spring 환경에서는 보통 Spring이 대신 관리

### DataSource

```xml
<dataSource type="POOLED">
```

DB 연결(Connection)을 관리하는 설정이다.

* POOLED : 커넥션 풀 사용
* DB 연결을 재사용하여 성능 향상
* 실무에서 가장 많이 사용

### Mapper 등록

```xml
<mapper resource="com/ohgiraffers/section01/xmlconfig/menu-mapper.xml"/>
```

MyBatis에게 SQL이 작성된 XML 파일의 위치를 알려준다.

## 3. ResultMap

DB 컬럼명과 DTO 필드명이 다를 때 매핑 규칙을 정의한다.

```xml
<id property="code" column="menu_code"/>
```

| DB 컬럼            | DTO 필드          |
| ---------------- | --------------- |
| menu_code        | code            |
| menu_name        | name            |
| menu_price       | price           |
| category_code    | categoryCode    |
| orderable_status | orderableStatus |

### id 태그

```xml
<id property="code" column="menu_code"/>
```

PK(Primary Key)를 나타낸다.

### result 태그

```xml
<result property="name" column="menu_name"/>
```

일반 컬럼을 매핑한다.

### parameterType

```xml

```

정수형 데이터를 전달받겠다는 의미이다.

### SQL

```sql

     , menu_name
     , menu_price
     , category_code
     , orderable_status
FROM tbl_menu
WHERE orderable_status = 'Y'
  AND menu_code = #{code}
```

### #{}

```xml
#{code}
```

MyBatis의 파라미터 바인딩 문법이다.

JDBC의

```java
PreparedStatement pstmt =
    con.prepareStatement(
        "SELECT * FROM tbl_menu WHERE menu_code = ?"
    );

pstmt.setInt(1, code);
```

와 동일한 역할을 수행한다.

### 실행 예시

```java
menuDAO.selectMenuByCode(sqlSession, 5);
```

↓

```sql
SELECT *
FROM tbl_menu
WHERE menu_code = 5
```

## 학습 정리

* `mybatis-config.xml`은 DB 연결 정보와 Mapper 등록을 담당한다.
* `resultMap`은 DB 컬럼명과 DTO 필드명을 연결해주는 매핑 규칙이다.
* `#{}`는 PreparedStatement의 `?`와 같은 역할을 수행한다.
* `selectOne()`은 단일 조회, `selectList()`는 다중 조회에 사용한다.
* MyBatis는 조회 결과를 자동으로 DTO 객체로 변환해준다.
* SqlSessionFactory는 애플리케이션에서 하나만 생성하고, SqlSession은 필요할 때마다 생성하여 사용한다.
