### JOIN

#### SELF JOIN
Self Join(셀프 조인)은 하나의 테이블이 두 가지 역할을 동시에 수행한다고 생각하면 이해가 빠릅니다.

지금 작성하신 코드를 예로 들어, 왜 이 구조가 필요한지 핵심만 짚어드릴게요.

1. 왜 테이블을 두 번 쓰나요? (핵심 개념)
데이터는 하나인데, 그 안에 '계급(상하 관계)'이 섞여 있기 때문입니다.

a (하위 역할): "나는 누구의 밑에 있는가?"를 찾는 테이블

b (상위 역할): "내가 누구의 상사(부모)인가?"를 보여주는 테이블

2. JOIN 조건 해석하기
ON a.ref_category_code = b.category_code

a.ref_category_code: 하위 카테고리가 들고 있는 "우리 부모님 번호"입니다.

b.category_code: 상위 카테고리가 들고 있는 "내 고유 번호"입니다.

쉽게 말해:
"내(a)가 들고 있는 '부모 번호'와 일치하는 '고유 번호'를 가진 데이터(b)를 옆에 붙여줘!"라는 뜻입니다.

#### SQL 실행 순서
SELECT - 어떤 테이블에서부터
FROM - 어떤 조건의 필터에 맞춰서
WHERE - 어떤 그룹으로 묶은 다음에
GROUP BY - 어떤 순서로 묶을지
HAVING - 어떤 조건을 가져올지
ORDER BY - 어떤 순서로 보여줄지

1. SELECT
```
# select
-- SELECT 컬럼명 FROM 테이블명;
-- 특정 데이터에서 원하는 데이터를 조회하는데 사용

-- 단일 열(컬럼) 데이터 검색
SELECT menu_name FROM tbl_menu;

-- 여러 열의 데이터 검색
SELECT

    menu_price
FROM
	tbl_menu;
    
-- 모든 열에서 데이터 검색( '*' : 모든 컬럼을 의미하는 와일드 카드)

-- FROM 절 없이 간단한 연산
SELECT 6 + 3;
SELECT 6 * 3;

-- 내장 함수 사용
SELECT NOW(); -- 현재 날짜와 시간을 보여줌
SELECT CONCAT('홍', '길동'); -- 여러 문자열을 합쳐준다.

-- 컬럼 별칭(Alias) 사용
SELECT CONCAT('홍', '길동') AS name ; -- 별칭
SELECT CONCAT('홍', '길동') AS 'Full name'; -- 별칭 띄워쓰기 시 '' 사용
```


3. ORDER BY
```
# ORDER BY 기준컬럼
-- SELECT 문의 가장 마지막에 위치하며, 결과 집합을 하나의 열로 정렬

-- 오름차순 정렬
SELECT 
	menu_code,
    menu_name,
    menu_price
FROM 
	tbl_menu
ORDER BY 
	menu_price ASC; -- ASC는 오름차순을 의미(생략 가능)
    
-- 내림차순 정렬
SELECT
	menu_code,
    menu_name,
    menu_price
FROM
	tbl_menu
ORDER BY
	menu_price DESC, -- 내림차순 (DESC 생략 불가능)
    menu_name ASC;   -- 2차 기준 : 가격이 같다면, 이름 오름차순으로

-- 연산 결과로 정렬(보통 별칭을 사용)
SELECT
	menu_code,
    menu_price,
    menu_code * menu_price AS calculated_value -- 곱한 결과를 select절 조회
FROM
	tbl_menu
ORDER BY
	menu_code * menu_price DESC;

-- FIELD(컬럼, 첫번째, 두번째, ...) : 컬럼 값이 목록의 몇 번째에 있는지 숫자로 알려준다.
SELECT FIELD('B', 'A', 'B', 'C');

-- 'N'을 1순위 'Y'를 2순위로 정렬
SELECT
	menu_name,
    orderable_status
FROM
	tbl_menu
ORDER BY
	FIELD(orderable_status, 'N', 'Y');

-- null 값이 있는 컬럼에 대한 정렬
SELECT
	category_code,
    category_name,
    ref_category_code
FROM
	tbl_category -- 카테고리 테이블에서 부터
ORDER BY
 -- ref_category_code IS NULL; -- NULL이면 TRUE(1), 아니면 FALSE(0) > NULL이 가장 아래 위치
 ref_category_code IS NULL DESC, -- NULL이 상단에 오도록 정렬
 ref_category_code DESC;
```

3. WHERE
```
# WHERE 조건
-- '조건 필더링'의 역할, 특정 조건에 맞는 레코드만을 선택하는데 사용
-- 1. 비교 연산자 활용(=, <>, >, <=)

SELECT
	menu_name,
    menu_price,
    orderable_status
FROM
	tbl_menu
WHERE
	orderable_status <> 'Y'; -- 해당 테이블과 값이 같지 않은 데이터 TRUE
SELECT
	menu_name,
    menu_price
FROM
	tbl_menu
WHERE
	menu_price < 13000;

-- 2. 논리 연산자(AND, OR)를 활용하여 조건 조합
SELECT
	menu_name,
    category_code,
    orderable_status
FROM
	tbl_menu
WHERE
	orderable_status = 'Y' AND category_code = 10;

-- OR(둘 중 하나라도 TRUE)
-- 가격이 만원 이상이고, 2만 5천원인 메뉴 찾기
SELECT
	menu_name,
    menu_price
FROM
	tbl_menu
WHERE
	menu_price >= 10000 AND menu_price <= 25000;
    
-- BETWEEN 연산자
SELECT
	menu_name,
    menu_price
FROM
	tbl_menu
WHERE
	-- 10000 이상 25000 이하(부정은 NOT을 붙인다)
	menu_price BETWEEN 10000 AND 250000;
    
-- LIKE
-- 메뉴 이름에 '마늘'이 포함된 메뉴 찾고 싶을 때, %는 0개 이상의 모든 문자를 의미하는 와일드 카드
SELECT
	menu_name
FROM
	tbl_menu
WHERE
	menu_name LIKE '%마늘%' ; 
    -- LIKE '%마늘' -- '마늘'로 끝나는 모든 것 (햇마늘, 깐마늘, 의성마늘 등)
    -- LIKE '마늘%' -- 중간에 '마늘'이 들어가는 모든 것

-- IN 연산자 : OR 연산자를 여러 번 쓰는 대신, 괄호 ( ) 안에 찾고 싶은 값들을 쉼표로 나열해서 사용
-- 카테고리 코드가 4번 OR 5번 OR 6번인 메뉴 찾기
SELECT
	menu_name,
    category_code
FROM
	tbl_menu
WHERE
	category_code IN (4,5,6);

-- IS NULL 연산자
-- 데이터베이스에서 NULL은 '0'이나 '빈 문자열'이 아닌, '값이 존재하지 않음'을 의미하는 상태이다. 
-- 상위 카테고리가 없는 (NULL) 카테고리 찾기
SELECT
	category_code,
    category_name,
    ref_category_code
FROM
	tbl_category
WHERE
	ref_category_code IS NULL; -- *부정 : IS NOT NULL
    
```

5. LIMIT
```
# LIMIT
-- 결과중에 딱 내가 원하는 마늠만 잘라서 줘! 라고 요청하는 키워드

-- LIMIT [개수]
SELECT
	menu_name,
    menu_price
FROM
	tbl_menu
ORDER BY
	menu_price DESC
LIMIT 5;

-- LIMIT [시작 위치(건너뛸 개수)]
-- 2번째 행부터 4개의 행을 가져오기
SELECT
	menu_code,
    menu_name,
    menu_price
FROM
	tbl_menu
ORDER BY
	menu_price DESC
LIMIT 1, 4;
```

6. JOIN
```
# JOIN

-- 1. 컬럼 별칭
SELECT
	menu_name AS '메뉴 이름',
    menu_price 
FROM
	tbl_menu;

-- 2. 테이블 별칭 (JOIN에서 필수)
SELECT
	menu_name,
    menu_price
FROM
	tbl_menu AS a;

-- INNER JOIN(교집합) - 가장 기본적인 조인, INNER 키워드 생략 가능
SELECT
	a.menu_name,
    b.category_name
FROM
	tbl_menu a
JOIN
	tbl_category b ON a.category_code = b.category_code;
    
SELECT *
FROM tbl_category;

-- OUTER JOIN (LEFT / RIGHT)

-- LEFT : 조인 구문을 기준으로 '왼쪽' 테이블의 데이터는 모두 보여주고,
-- 오른쪽 테이블에서 짝이 맞는 데이터가 없으면 'NULL'로 채워 보여준다.
-- RIGHT : 기준이 오른쪽 테이블이다.
SELECT
	a.category_name,
    b.menu_name
FROM
	tbl_category a
LEFT JOIN tbl_menu b ON a.category_code = b.category_code;

-- SELF JOIN
SELECT
	a.category_name AS '하위 카테고리',
    b.category_name AS '상위 카테고리'
FROM
	tbl_category a -- 하위 역할
JOIN tbl_category b ON a.ref_category_code = b.category_code; -- 상위 역할   

```
